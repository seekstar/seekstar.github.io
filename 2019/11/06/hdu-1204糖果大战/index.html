double eps = 1e-8;
int sgn(double x) {
    return x < -eps ? -1 : (x > eps ? 1 : 0);
}

int main() {
    double n, m;
    double p, q;

    while (~scanf("%lf%lf%lf%lf", &n, &m, &p, &q)) {
        if (sgn(n) == 0) {
            puts("0.00");
        } else if (sgn(m) == 0) {
            puts("1.00");
        } else {
            double pb = q * (1 - p), pa = (p * (1 - q));
            if (sgn(pa) == 0) {
                puts("0.00");
            } else if (sgn(pb) == 0) {
                puts("1.00");
            } else {
                double t = pb / pa;
                if (sgn(t - 1)) {
                    printf("%.2f\n", (pow(t, n) - 1) / (pow(t, n + m) - 1));
                } else {
                    printf("%.2f\n", n / (n + m));
                }
            }
        }
    }
    return 0;
}
