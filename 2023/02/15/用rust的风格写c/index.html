tag_invokeеҸҜд»Ҙи§ЈеҶізәҜCPOзҡ„е‘ҪеҗҚеҶІзӘҒпјҢжҖқи·ҜжҳҜи®©жүҖжңүжҺҘеҸЈйғҪйҖҡиҝҮtag_invokeе®һзҺ°пјҢиҖҢи°ғз”Ёж—¶йҖҡиҝҮtagжқҘжҢҮе®ҡе…·дҪ“иҰҒи°ғз”Ёе“ӘдёӘе®һзҺ°гҖӮдёҠйқўзҡ„дҫӢеӯҗз”Ёtag_invoke + CPOйҮҚеҶҷпјҡ

#include <iostream>
#include <vector>
namespace lib {
namespace detail {
struct print_t {
	template <typename T>
	void operator()(T x) {
		tag_invoke(print_t{}, x);
	}
};
template <typename T>
void tag_invoke(print_t, T x) {
	std::cout << x;
}
template <typename T>
void tag_invoke(print_t, const std::vector<T>& v) {
	for (const auto& x : v) {
		tag_invoke(print_t{}, x);
	}
}
} // namespace detail
template <auto& Tag>
using tag_t = std::decay_t<decltype(Tag)>;

	out << a.x;
	return out;
}

	std::cout << "{x:" << a.x << "}";
}
int main() {

	lib::print(v);
	std::cout << std::endl;
	lib::print(A{2});

}
